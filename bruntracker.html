<div class="card">

  <h3>運動記錄</h3>

  <select id="activity">

    <option value="8">跑步</option>

    <option value="6">重訓</option>

    <option value="5">快走</option>

  </select><br>

  <input type="number" id="time" placeholder="分鐘">

  <br>

  <button onclick="addRecord()">新增</button>

</div>


<div class="card">

  <h3>今日紀錄</h3>

  <div id="records"></div>

  <button onclick="finishDay()">完成🔥</button>

</div>


<script>

let records = [];

let totalKcal = 0;

let weight = 70;


function addRecord(){

  let met = document.getElementById('activity').value;

  let time = document.getElementById('time').value;


  if(!time) return alert('請輸入時間');


  let kcal = weight * (time/60) * met;

  kcal = Math.round(kcal);


  records.push(kcal);

  document.getElementById('records').innerHTML += `<div>${kcal} kcal</div>`;

}


function finishDay(){

  let sum = records.reduce((a,b)=>a+b,0);

  totalKcal += sum;


  document.getElementById('kcal').innerText = totalKcal + ' kcal';


  let fire = document.getElementById('fire');

  fire.className = 'fire';


  if(totalKcal > 600) fire.classList.add('huge');

  else if(totalKcal > 300) fire.classList.add('big');


  records = [];

  document.getElementById('records').innerHTML = '';

}

</script>
